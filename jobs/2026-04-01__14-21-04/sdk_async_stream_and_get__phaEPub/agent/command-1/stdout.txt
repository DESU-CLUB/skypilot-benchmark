resources = sky.Resources(cpus=2)           # 2 CPUs, no GPU

    setup="echo 'setup done'",

request_id = sky.launch(                    # async call → returns request_id
    task, cluster_name='eval-cluster'
)

job_id, handle = sky.stream_and_get(        # block + stream logs until done
    request_id
